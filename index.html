WordPress Multilingual UX — Google Language Translator (JS excerpt)

/* custom.js — фрагмент реальної логіки мультимовності (скорочено) */
(function initLanguageMenus() {
  // Зчитуємо поточну мову з cookie googtrans (fallback: uk)
  function getCookieLang() {
    const m = document.cookie.match(/(?:^|;\s*)googtrans=([^;]+)/);
    if (!m) return 'uk';
    const parts = decodeURIComponent(m[1]).split('/');
    return parts[parts.length - 1] || 'uk';
  }

  // Записуємо cookie у кількох варіантах (plain/encoded + domain), щоб GT стабільно підхоплював мову
  function setGoogTransCookie(lang) {
    const plain = '/uk/' + lang;
    const enc = encodeURIComponent(plain);

    function writeCookie(val, domain) {
      let s = 'googtrans=' + val + '; path=/; SameSite=Lax';
      if (domain) s += '; domain=' + domain;
      document.cookie = s;
    }

    writeCookie(plain);
    writeCookie(enc);

    const host = location.hostname;
    if (host && host.indexOf('.') !== -1) {
      writeCookie(plain, host);
      writeCookie(enc, host);
      writeCookie(plain, '.' + host);
      writeCookie(enc, '.' + host);
    }
  }

  // Старт перекладу (GT)
  function forceTranslate(lang) {
    setGoogTransCookie(lang);
    if (window.doGTranslate) return window.doGTranslate('uk|' + lang);

    const sel = document.querySelector('.goog-te-combo');
    if (sel) { sel.value = lang; sel.dispatchEvent(new Event('change')); }
  }

  // Клік по пункту мови → ставимо cookie → запускаємо GT → оновлюємо наші правки
  document.addEventListener('click', (e) => {
    const a = e.target.closest && e.target.closest('a.single-language');
    if (!a) return;

    e.preventDefault();
    const lang = a.classList.contains('en') ? 'en'
               : a.classList.contains('de') ? 'de'
               : a.classList.contains('ru') ? 'ru'
               : 'uk';

    forceTranslate(lang);

    // Після перемикання — підміни текстів/обмеження notranslate/flatten breadcrumb тощо
    window.__mlOnLangSwitch && window.__mlOnLangSwitch();
  });
})();

// Elementor може «перемальовувати» блоки — підхоплюємо ререндери і повторюємо наші правки
(function mlObserveRerenders() {
  const root = document.body;
  if (!root || window.__mlNtObserver) return;

  window.__mlNtObserver = new MutationObserver(() => {
    clearTimeout(window.__mlNtObserverT);
    window.__mlNtObserverT = setTimeout(() => {
      window.__mlOnLangSwitch && window.__mlOnLangSwitch();
    }, 140);
  });

  window.__mlNtObserver.observe(root, { childList: true, subtree: true });
})();

Google Sheets Automation (Apps Script)

// Copy selected rows to the summary sheet with formatting and totals
function copyToSummarySheet(sheet, totalRow) {
  const ss = SpreadsheetApp.getActiveSpreadsheet();
  const summary = ss.getSheetByName("Sheet4");
  const start = 2;
  const numRows = totalRow - 1;
  const range = sheet.getRange(start, 1, numRows, 6);
  range.copyTo(summary.getRange(summary.getLastRow() + 3, 1));
  const lastRow = summary.getLastRow();
  const table = summary.getRange(lastRow - numRows + 1, 1, numRows, 14);
  table.setBorder(true, true, true, true, true, true);
  summary.getRange(lastRow + 1, 14)
         .setValue(`=SUM(N${lastRow - numRows + 1}:N${lastRow})`)
         .setFontWeight("bold");
}
